198    inplace: bool = False,
199) -> np.ndarray:
200    """
201    Adds random noise seperately to each pixel of the image. The noise works
202    for both channel first and last images. Follows a uniform distribution.
203    Input should be (height, width, channels) or (channels, height, width).
204
205    NOTE: Beware of datatypes. Consider casting to float32 before adding noise.
206    
207    Parameters
208    ----------
209    X : np.ndarray
210        The image to add noise to.
211
212    max_amount : float, optional
213        The maximum amount of noise to add, sampled uniformly, default: 0.1.
214
215    additive : bool, optional
216        Whether to add or multiply the noise, default: False.
217
218    channel_last : bool, optional
219        Whether the image is (channels, height, width) or (height, width, channels), ignored for this function.
220        Kept to keep the same function signature as other augmentations, default: True.
221    
222    inplace : bool, optional
223        Whether to perform the rotation in-place, default: False.
224
225    Returns
226    -------
227    np.ndarray
228        The image with uniform noise.
229    """
230    if not inplace:
231        X = X.copy()
232
233    amount = np.random.rand() * max_amount
234
235    if additive:
236        random_noise = np.random.uniform(-amount, amount, X.shape).astype(X.dtype)
237        X[:] += random_noise
238    else:
239        random_noise = np.random.uniform(1 - amount, 1 + amount, X.shape).astype(X.dtype)
240        X[:] *= random_noise
241
242    return X

360@jit(nopython=True, nogil=True, cache=True, fastmath=True, parallel=True)
361def augmentation_contrast(
362    X: np.ndarray,
363    max_amount: float = 0.1,
364    channel_last: bool = True,
365    inplace: bool = False,
366) -> np.ndarray:
367    """
368    Changes the contrast of an image by a random amount, seperately for each channel.
369    Input should be (height, width, channels) or (channels, height, width).
370    
371    NOTE: Beware of datatypes. Consider casting to float32 before adding noise.
372
373    Parameters
374    ----------
375    X : np.ndarray
376        The image to change the contrast of.
377
378    max_amount : float, optional
379        The max amount to change the contrast by, default: 0.1.
380
381    channel_last : bool, optional
382        Whether the image is (channels, height, width) or (height, width, channels), default: True.
383    
384    inplace : bool, optional
385        Whether to perform the rotation in-place, default: False.
386
387    Returns
388    -------
389    np.ndarray
390        The image with changed contrast.
391    """
392    if not inplace:
393        X = X.copy()
394
395    amount = np.random.rand() * max_amount
396
397    channels = X.shape[2] if channel_last else X.shape[0]
398    mean_pixel = np.zeros(channels, dtype=np.float32)
399
400    # Numba workaround because numba does not support axis argument in np.mean
401    if channel_last:
402        for i in prange(channels):
403            mean_pixel[i] = np.nanmean(X[:, :, i])
404    else:
405        for i in prange(channels):
406            mean_pixel[i] = np.nanmean(X[i, :, :])
407
408    for i in prange(channels):
409        X[:, :, i] = (X[:, :, i] - mean_pixel[i]) * (1 + amount) + mean_pixel[i]
410
411    return X

423    Drops random pixels from an image. Input should be (height, width, channels) or (channels, height, width).
424    Only drops pixels from features, not labels.
425    
426    Parameters
427    ----------
428    X : np.ndarray
429        The image to drop a pixel from.
430
431    drop_probability : float, optional
432        The probability of dropping a pixel, default: 0.05.
433
434    drop_value : float, optional
435        The value to drop the pixel to, default: 0.0.
436
437    channel_last : bool, optional
438        Whether the image is (channels, height, width) or (height, width, channels), default: True.
439    
440    inplace : bool, optional
441        Whether to perform the rotation in-place, default: False.
442
443    Returns
444    -------
445    np.ndarray
446        The image with dropped pixels.
447    """
448    if not inplace:
449        X = X.copy()
450
451    # Create a mask of random numbers
452    mask = np.random.uniform(0.0, 1.0, size=X.shape).astype(np.float32)
453
454    # Agreed. This looks terrible. But it's the only way to get numba to parallelize this.
455    if channel_last:
456        for col in prange(X.shape[0]):
457            for row in prange(X.shape[1]):
458                for channel in prange(X.shape[2]):
459                    if mask[col, row, channel] <= drop_probability:
460                        X[col, row, channel] = drop_value
461
462    else:
463        for channel in prange(X.shape[0]):
464            for col in prange(X.shape[1]):
465                for row in prange(X.shape[2]):
466                    if mask[channel, col, row] <= drop_probability:
467                        X[channel, col, row] = drop_value
468
469    return X

833    two images and then adding a random weight to each image.
834
835    Mixup involves taking two images and blending them together by randomly interpolating
836    their pixel values. More specifically, suppose we have two images x and x' with their
837    corresponding labels y and y'. To generate a new training example, mixup takes a
838    weighted sum of x and x', such that the resulting image x^* = λx + (1-λ)x',
839    where λ is a randomly chosen interpolation coefficient. The label for the new image
840    is also a weighted sum of y and y' based on the same interpolation coefficient.
841
842    input should be (height, width, channels) or (channels, height, width).
843
844    Parameters
845    ----------
846    X_target : np.ndarray
847        The image to transfer to.
848
849    y_target : np.ndarray
850        The label to transfer to.
851
852    X_source : np.ndarray
853        The image to transfer from.
854
855    y_source : np.ndarray
856        The label to transfer from.
857
858    min_size : float, optional
859        The minimum mixup coefficient, default: 0.333.
860
861    max_size : float, optional
862        The maximum mixup coefficient, default: 0.666.
863
864    label_mix : int, optional
865        If 0, the labels will be mixed by the weights. If 1, the target label will be used. If 2, 
866        the source label will be used. If 3, the max of the labels will be used. If 4, the min 
867        of the labels will be used. If 5, the max of the image with the highest weight will be used. 
868        If 6, the min of the image with the highest weight will be used. If 7, the sum of the labels 
869        will be used, default: 0.
870
871    channel_last : bool, optional
872        Whether the image is (channels, height, width) or (height, width, channels), default: True.
873    
874    inplace : bool, optional
875        Whether to perform the rotation in-place, default: False.
876
877    Returns
878    -------
879    Tuple[np.ndarray, np.ndarray]
880        The mixed up image and label.
881    """
882    if not inplace:
883        X_target = X_target.copy()
884        y_target = y_target.copy()
885
886    mixup_coeff = np.float32(
887        min(np.random.uniform(min_size, max_size + 0.001), 1.0),
888    )
889
890    X_target[:] = (X_target * mixup_coeff) + (X_source * (np.float32(1.0) - mixup_coeff))
891
892    if label_mix == 0:
893        y_target[:] = (y_target * mixup_coeff) + (y_source * (np.float32(1.0) - mixup_coeff))
894    elif label_mix == 1:
895        y_target[:] = y_target # pylint: disable=self-assigning-variable
896    elif label_mix == 2:
897        y_target[:] = y_source
898    elif label_mix == 3:

900    elif label_mix == 4:

902    elif label_mix == 5:

904    elif label_mix == 6:
905        y_target[:] = y_target if mixup_coeff >= 0.5 else y_source
906    elif label_mix == 7:
907        y_target[:] = y_target + y_source
908
909    return X_target, y_target
